Function Get-DadJoke {
    [CmdletBinding()]
    Param()

    $uri = 'https://icanhazdadjoke.com'
    $get = Invoke-RestMethod $uri -Headers @{accept = 'application/json' } -DisableKeepAlive

    [PSCustomObject]@{
        PSTypeName = 'PSDadJoke'
        Joke       = $get.Joke.Trim()
        Permalink  = "$uri/j/$($get.ID)"
        ID         = $get.ID
    }
}
